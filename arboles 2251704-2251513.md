class Nodo:
    def __init__(self, dato):
        self.dato = dato
        self.izquierda = None
        self.derecha = None

class Arbol:
    def __init__(self):
        self.raiz = None

    
    def insertar(self, dato):
        nuevo = Nodo(dato)

        if self.raiz is None:
            self.raiz = nuevo
            return

        actual = self.raiz

        while True:
            if dato < actual.dato:
                if actual.izquierda is None:
                    actual.izquierda = nuevo
                    break
                else:
                    actual = actual.izquierda
            else:
                if actual.derecha is None:
                    actual.derecha = nuevo
                    break
                else:
                    actual = actual.derecha

    
    def peso(self, nodo):
        if nodo is None:
            return 0
        return 1 + self.peso(nodo.izquierda) + self.peso(nodo.derecha)

    
    def altura(self, nodo):
        if nodo is None:
            return 0

        altura_izq = self.altura(nodo.izquierda)
        altura_der = self.altura(nodo.derecha)

        if altura_izq > altura_der:
            return altura_izq + 1
        else:
            return altura_der + 1

    
    def orden(self, nodo):
        if nodo is None:
            return 0

        hijos = 0
        if nodo.izquierda is not None:
            hijos += 1
        if nodo.derecha is not None:
            hijos += 1

        orden_izq = self.orden(nodo.izquierda)
        orden_der = self.orden(nodo.derecha)

        return max(hijos, orden_izq, orden_der)




arbol = Arbol()

cantidad = int(input("Cuantos nodos quiere ingresar: "))

for i in range(cantidad):
    valor = int(input("Ingrese el valor del nodo: "))
    arbol.insertar(valor)

print("\nRESULTADOS")

print("Peso del arbol:", arbol.peso(arbol.raiz))
print("Altura del arbol:", arbol.altura(arbol.raiz))
print("Orden del arbol:", arbol.orden(arbol.raiz))
